6,                # первый аргумент 
7,                # второй аргумент
-> multiplecate   # вызов
=> return         # выход.

# В коде запятые обозначают буквально: "запомнить аргумент"



def multiplecate :fn  # функция multiplecate. 
  param               # специальный блок обьявления параметров. Синтаксис такой же как у ram.
    first_arg,        # первый параметр 
    second_arg,       # второй параметр

  first_arg' * second_arg' => return


# немного о запятых:
#     такие блоки как "param" и "stack" служат для описания данных и не подразумевают собой выполнение кода.
#     Поэтому запятые можно использовать в них как разделитель. 
#     блок "ram" может содержать вычесления, но они выполняются на этапе компиляции и не подразумевают вызов функции.
#     в нутри блока params запятая просто подсказывает компилятору что вы закончили описание очередного параметра. 
#     её необходимость станет чуть более очевидна когда мы добавим в код типы данных. 
#     Или например если скопировать их и использовать в качестве аргументов - в этом случае вам бы всё равно пришлось везде расставить запятые...
