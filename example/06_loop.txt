def get_factorial :fn
  params max,
  locals f, n,

  1 => f  
  0 => n
  loop 
    cmp n' ?? max'
      >  break
    n' + 1  => n
    f' * n' => f
    continue

  f' => return

# loop     - блок "цикл" с неявным переходом из его конца в его же начало.
# break    - выход из цикла
# continue - комманда перехода на начало цикла. Здесь указана только для явности.
